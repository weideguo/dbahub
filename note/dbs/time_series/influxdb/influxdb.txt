# 启动
influxd run -config <path>

# 备份
influxd backup -portable -db example-db  /path/to/backup-directory
# 还原
influxd restore -portable -db example-db /path/to/backup-directory

# 操作完毕后不需要重启
# 可能出现error updating meta？


# 备份
influxd backup -db example-db  /path/to/backup-directory
# 还原
# 导入时库需要存在，且设置RETENTION POLICY跟备份的一致，influxdb需要处于关闭状态
influxd restore                       \
-db example-db -newdb example-db      \    # 即使同一个也有设置？
-datadir /path/to/influxdb-data-dir   \
-metadir /path/to/influxdb-meta-dir   \    # 这个会导致其他的库都不可见！
/path/to/backup-directory


# 查看默认配置
influxd config

# 导出成csv
influx -database 'your_database' -execute 'SELECT * FROM "your_measurement"' -format csv > output.csv


端口
8086 客户端连接，http
8088 备份、恢复数据的RPC

可以直接通过http访问
query?db=my_db&q=select...

curl -G "http://localhost:8086/query?db=mydb" \
  -u my_account:my_secure_password \
  --data-urlencode "q=SELECT * FROM my_measurement"

# 插入多条记录
curl -i -XPOST "http://localhost:8086/write?db=mydb20250728" \
--data-binary '
cpu,host=server01,region=us-west value=0.11,value1=0.12 1500000001000000000
cpu,host=server02,region=us-east value=0.22,value1=0.23 1500000002000000000
memory,host=server01 used_percent=99.0 1500000001000000000
'

# curl直接读取文件
--data-binary @data.lp
