
tags   有索引
fields 无索引
point  记录
series 不同的tags组合，不应该过大，过大会导致内存使用过大
 
tb_name,tag1=v01,tag2=v02,...                                    # series示例
tb_name,tag1=v11,tag2=v12,...                                    #

SHOW SERIES FROM tb_name;                                        # 查看series的详细信息
SHOW SERIES CARDINALITY FROM tb_name;                            # 查看series的数量
                                                                 
SHOW TAG KEYS FROM tb_name;                                      # 查看哪些字段是tag，即查看有哪些tag是series的一部分
SHOW TAG VALUES CARDINALITY FROM tb_name WITH KEY = tag_name;    # 查看series中这个tag不重复的个数

SHOW FIELD KEYS FROM tb_name;                                    # 查看哪些字段是field
